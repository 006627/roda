* Solr HTTP (create cores from RODA install dir in /home/hsilva/.roda/)

	$> bin/solr start
	$> for i in $(ls /home/hsilva/.roda/config/index/);do if [ -d /home/hsilva/.roda/config/index/$i ]; then bin/solr create -c $i -d /home/hsilva/.roda/config/index/$i/conf/ ; fi; done
